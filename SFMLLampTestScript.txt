#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Transmits commands over the raw system connection and catches the text output line
std::string talkToHardware(const std::string& command) {
    // One-time serial interface initialization setup
    static bool portConfigured = false;
    if (!portConfigured) {
        system("stty -F /dev/ttyACM0 115200 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");
        system("stty -F /dev/ttyACM1 115200 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");
        portConfigured = true;
    }

    // Open active hardware port line mapping
    std::fstream serialPort("/dev/ttyACM0", std::ios::in | std::ios::out);
    if (!serialPort.is_open()) {
        serialPort.open("/dev/ttyACM1", std::ios::in | std::ios::out);
        if (!serialPort.is_open()) return "";
    }

    // Direct firmware message passing
    serialPort << command << "\r\n";
    serialPort.flush();

    std::string response = "";
    std::getline(serialPort, response);
    return response;
}

// Inspects the state of physical Input 0 (Pin 3 on your ModBay)
bool isPhysicalButtonPressed() {
    std::string state = talkToHardware("op dio-state");
    
    // Safety exit if communication drops out or string isn't fully formed
    if (state.length() < 8) return false;
    
    // FIXED: Accessing index [0] parses the specific character character '0' (Pressed/GND)
    return (state[0] == '0');
}

// Drives physical Output 0 (Pin 4 on your ModBay) high or low
void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        talkToHardware("op set-output 0 1");
    } else {
        talkToHardware("op set-output 0 0");
    }
}

int main() {
    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon 400 ModBay IO Bridge");
    window.setFramerateLimit(30); 

    // Setup visual window elements
    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); // Default state: Open

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Assure pins clear on start

    std::cout << "Application online. Click the visual window area to toggle Pin 4 (Out0)." << std::endl;

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // WINDOW MOUSE INTERACTION: Toggles Pin 4 (Out0) status
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "Sent toggle signal. Lamp target is: " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE PIN SURVEY: Read Pin 3 (In0) to toggle UI graphic colors
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Active Connection
        } else {
            visualLamp.setFillColor(sf::Color::Red);   // Standby
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Clear power paths upon exit
    return 0;
}
