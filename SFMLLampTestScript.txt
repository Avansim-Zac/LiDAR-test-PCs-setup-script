#include <SFML/Graphics.hpp>
#include <iostream>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

// OnLogic Kernel Architecture Payload Structure Models
struct pse_ioctl_do_data {
    uint8_t index; // The Output Pin index target (0 for Out0/Pin 4)
    uint8_t state; // Target configuration state (1 = HIGH, 0 = LOW)
} __attribute__((packed));

struct pse_ioctl_di_data {
    uint8_t index; // Pin index lookup
    uint8_t state; // Return slot variable for pin readings
} __attribute__((packed));

// Re-calculated Low-level Linux ioctl Commands Matching Kernel Architecture
#define PSE_IOCTL_MAGIC 'p'
#define PSE_IOCTL_GET_DI _IOWR(PSE_IOCTL_MAGIC, 3, struct pse_ioctl_di_data)
#define PSE_IOCTL_SET_DO _IOW(PSE_IOCTL_MAGIC, 4, struct pse_ioctl_do_data)

// Queries ModBay Pin 3 (In0)
bool isPhysicalButtonPressed() {
    int fd = open("/dev/pse0", O_RDWR);
    if (fd < 0) return false;

    struct pse_ioctl_di_data data;
    data.index = 0; // We check Input Channel 0 (Pin 3)
    data.state = 1; // Default backstop state initialization

    if (ioctl(fd, PSE_IOCTL_GET_DI, &data) < 0) {
        close(fd);
        return false;
    }
    close(fd);

    // If data.state reads back 0, Pin 3 has bridged to Pin 2 (GND)
    return (data.state == 0);
}

// Controls ModBay Pin 4 (Out0)
void setPhysicalLamp(bool turnOn) {
    int fd = open("/dev/pse0", O_RDWR);
    if (fd < 0) return;

    struct pse_ioctl_do_data data;
    data.index = 0; // Target Output 0 (Pin 4)
    data.state = turnOn ? 1 : 0;

    ioctl(fd, PSE_IOCTL_SET_DO, &data);
    close(fd);
}

int main() {
    std::cout << "Loading Aligned ModBay System Controller Context..." << std::endl;

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay IO Test");
    window.setFramerateLimit(30); 

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); 

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Assure pins start dead

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // SCREEN INTERACTION DETECTED
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "Dispatched ModBay ioctl State Request -> " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE EVENT SURVEY: Continually evaluate button to override color states
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical circuit closed (Button Down)
        } else {
            // Updated: Click feedback is decoupled from ioctl fails so the shape updates immediately
            if (physicalLampState) {
                visualLamp.setFillColor(sf::Color::Blue);  // UI click active (Output High)
            } else {
                visualLamp.setFillColor(sf::Color::Red);   // Standby (Output Low)
            }
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Safeguard shutdown sequence
    return 0;
}
