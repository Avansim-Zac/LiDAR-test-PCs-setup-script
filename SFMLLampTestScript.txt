#include <SFML/Graphics.hpp>
#include <iostream>
#include "pse.h" // The driver's internal header file

// Queries Pin 3 (In0) using the real driver framework
bool isPhysicalButtonPressed(int fd) {
    if (fd < 0) return false;

    uint8_t states = 0xFF;
    // Reads all input pins into a single bitwise flag block
    if (pse_get_di(fd, &states) != 0) {
        return false;
    }

    // Pin 3 (In0) maps to bit 0. It drops to '0' when shorted to Pin 2 (GND)
    return ((states & 0x01) == 0);
}

int main() {
    std::cout << "Opening communication channel with Intel PSE Driver..." << std::endl;

    // MANDATORY STEP: Open and bind the ISHTP client interface connection
    int fd = pse_open();
    if (fd < 0) {
        std::cerr << "CRITICAL ERROR: Failed to talk to /dev/pse0. Run script as sudo!" << std::endl;
        return 1;
    }

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay IO Test");
    window.setFramerateLimit(30);

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red);

    bool physicalLampState = false;
    pse_set_do(fd, 0, 0); // Force Output 0 (Pin 4) Low on startup

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }

            // SCREEN CLICK EVENT: Toggles physical Output 0 (Pin 4)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    
                    // Directly writes 1 (HIGH) or 0 (LOW) to ModBay Output 0 (Pin 4)
                    pse_set_do(fd, 0, physicalLampState ? 1 : 0);
                    std::cout << "Sent Signal to Pin 4 -> " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE POLLING: Check the physical button state
        if (isPhysicalButtonPressed(fd)) {
            visualLamp.setFillColor(sf::Color::Green); // Physical button pressed
        } else {
            // Decay back to screen toggle states if button is released
            if (physicalLampState) {
                visualLamp.setFillColor(sf::Color::Blue);  // Screen click active (Output High)
            } else {
                visualLamp.setFillColor(sf::Color::Red);   // Standby (Output Low)
            }
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    // Clean up pins and safely close the driver channel upon exit
    pse_set_do(fd, 0, 0);
    pse_close(fd);
    return 0;
}
