#include <SFML/Graphics.hpp>
#include <iostream>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

// OnLogic Kernel Driver Layout Models
struct pse_ioctl_do_data {
    uint8_t index; // Target pin line index (0 for Out0)
    uint8_t state; // Electrical configuration parameter (1 = HIGH, 0 = LOW)
} __attribute__((packed));

struct pse_ioctl_di_data {
    uint8_t index; // Pin lookup mapping target (0 for In0)
    uint8_t state; // Return buffer target from physical registers
} __attribute__((packed));

// Precise Linux ioctl Magic Packets Matching OnLogic Kernel Modules
#define PSE_IOCTL_MAGIC 'p'
#define PSE_IOCTL_GET_DI _IOWR(PSE_IOCTL_MAGIC, 3, struct pse_ioctl_di_data)
#define PSE_IOCTL_SET_DO _IOW(PSE_IOCTL_MAGIC, 4, struct pse_ioctl_do_data)

// Queries Pin 3 (In0) state
bool isPhysicalButtonPressed() {
    int fd = open("/dev/pse0", O_RDWR);
    if (fd < 0) return false;

    struct pse_ioctl_di_data ioctlPayload;
    ioctlPayload.index = 0; // Check Input 0
    ioctlPayload.state = 1; // Default fallback state value

    if (ioctl(fd, PSE_IOCTL_GET_DI, &ioctlPayload) < 0) {
        close(fd);
        return false;
    }
    close(fd);

    // ModBay inputs read '0' when the physical button completes the circuit to Ground
    return (ioctlPayload.state == 0);
}

// Controls Pin 4 (Out0) state
void setPhysicalLamp(bool turnOn) {
    int fd = open("/dev/pse0", O_RDWR);
    if (fd < 0) return;

    struct pse_ioctl_do_data ioctlPayload;
    ioctlPayload.index = 0; // Target Output 0
    ioctlPayload.state = turnOn ? 1 : 0;

    ioctl(fd, PSE_IOCTL_SET_DO, &ioctlPayload);
    close(fd);
}

int main() {
    std::cout << "Loading Direct Linux ioctl Controller Channel..." << std::endl;

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay Native IO Bridge");
    window.setFramerateLimit(30);

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); // Default state: Off

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Clear pin output lines on launch

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }

            // WINDOW MOUSE CLICK EVENT: Toggle the physical lamp on Pin 4 (Out0)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "UI Registered Click. Toggled State to -> " << physicalLampState << std::endl;
                }
            }
        }

        // HARDWARE EVENT POLLING: Check Pin 3 (In0) to update window layouts dynamically
        // Decoupled color assignment logic ensures the window remains responsive
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical button pressed
        } else {
            if (physicalLampState) {
                visualLamp.setFillColor(sf::Color::Blue);  // UI click active (Output High)
            } else {
                visualLamp.setFillColor(sf::Color::Red);   // Standby (Output Low)
            }
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Safeguard shutdown sequence on exit
    return 0;
}
