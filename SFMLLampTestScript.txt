#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Transmits plain-text commands to the Karbon microcontroller serial node
std::string talkToMCU(const std::string& command) {
    // Ubuntu maps the Karbon serial terminal port to ttyACM1
    std::fstream serialPort("/dev/ttyACM1", std::ios::in | std::ios::out);
    if (!serialPort.is_open()) {
        // Fallback to ACM0 if port numbering swapped on boot
        serialPort.open("/dev/ttyACM0", std::ios::in | std::ios::out);
        if (!serialPort.is_open()) return "";
    }

    // Per documentation: "Commands are expected to be sent by line transmitted in a single transaction"
    serialPort << command << "\n";
    serialPort.flush();

    // Read the line of text echoed back by the MCU firmware
    std::string response = "";
    std::getline(serialPort, response);
    return response;
}

// Queries Pin 3 (In0) state
bool isPhysicalButtonPressed() {
    // Official documentation command string: "dio-state"
    std::string state = talkToMCU("dio-state");
    
    // Safety exit if communication drops out or responds with blank lines
    if (state.length() < 8) return false;
    
    // Index 0 represents Input 0 (Pin 3 on your ModBay).
    // It reads character '1' normally, and drops to character '0' when shorted to Ground.
    return (state[0] == '0');
}

// Controls Pin 4 (Out0) state
void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        // Official command: "set-do 1---" (Sets Output 0 to High, leaves O1, O2, O3 unchanged)
        talkToMCU("set-do 1---");
    } else {
        // Official command: "set-do 0---" (Sets Output 0 to Low, leaves O1, O2, O3 unchanged)
        talkToMCU("set-do 0---");
    }
}

int main() {
    std::cout << "Connecting to Karbon Microcontroller Virtual Serial Port..." << std::endl;

    // One-time port initialisation: Set the line to raw mode so Ubuntu doesn't filter text
    // The documentation mentions it supports various baud rates natively
    std::system("stty -F /dev/ttyACM1 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");
    std::system("stty -F /dev/ttyACM0 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon 400 Final IO Test");
    window.setFramerateLimit(30); // 30Hz limit provides snappy serial polling responses

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); // Default state: Off

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Assure pins clear on application launch

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }

            // WINDOW MOUSE CLICK EVENT: Toggle the physical lamp on Pin 4 (Out0)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "UI Clicked. Sent command to toggle Out0 -> " 
                              << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE EVENT POLLING: Check Pin 3 (In0) to update window layouts dynamically
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical button completes circuit to GND
        } else {
            if (physicalLampState) {
                visualLamp.setFillColor(sf::Color::Blue);  // UI Click active (Lamp is illuminated)
            } else {
                visualLamp.setFillColor(sf::Color::Red);   // Standby (Lamp is off)
            }
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Safeguard shutdown sequence on exit
    return 0;
}
