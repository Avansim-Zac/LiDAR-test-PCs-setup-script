#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Raw firmware hex byte frames parsed natively by the Intel PSE ModBay controller
const uint8_t CMD_READ_DIO[]   = { 0x01, 0x00 }; 
const uint8_t CMD_WRITE_HIGH[] = { 0x02, 0x00, 0x01 }; // Set Out0 (Pin 4) On
const uint8_t CMD_WRITE_LOW[]  = { 0x02, 0x00, 0x00 }; // Set Out0 (Pin 4) Off

// Sends a raw hex buffer packet down the internal system bus mapping
bool transmitBusPayload(const uint8_t* command, size_t size) {
    // Standard Linux paths where Intel registers the Elkhart Lake PSE controller nodes
    std::ofstream device("/dev/ishtp-pse", std::ios::binary);
    if (!device.is_open()) {
        device.open("/dev/mei0", std::ios::binary); // Alternative Intel bus layer mapping
        if (!device.is_open()) return false;
    }
    
    device.write(reinterpret_cast<const char*>(command), size);
    device.flush();
    return true;
}

// Queries Pin 3 (In0) to see if it is bridged to Pin 2 (GND)
bool isPhysicalButtonPressed() {
    std::fstream device("/dev/ishtp-pse", std::ios::in | std::ios::out | std::ios::binary);
    if (!device.is_open()) {
        device.open("/dev/mei0", std::ios::in | std::ios::out | std::ios::binary);
        if (!device.is_open()) return false;
    }

    device.write(reinterpret_cast<const char*>(CMD_READ_DIO), sizeof(CMD_READ_DIO));
    device.flush();

    char responseByte = 0;
    device.read(&responseByte, 1);

    // Bit index 0 tracks Input 0. If it reads 0, the button circuit is closed.
    return ((responseByte & 0x01) == 0);
}

void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        transmitBusPayload(CMD_WRITE_HIGH, sizeof(CMD_WRITE_HIGH));
    } else {
        transmitBusPayload(CMD_WRITE_LOW, sizeof(CMD_WRITE_LOW));
    }
}

int main() {
    std::cout << "Initializing Native Bus Interaction Loop..." << std::endl;

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay IO Test");
    window.setFramerateLimit(30); 

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red);

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); 

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // SCREEN CLICK EVENT: Toggles physical Output 0 (Pin 4)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "Screen Clicked. Output Pin 4 Target: " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE POLLING LOOP: Reads Pin 3 to control window layout colors
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green);
        } else {
            visualLamp.setFillColor(sf::Color::Red);
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); 
    return 0;
}
