#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Opens the raw system connection, posts a command string, and catches the feedback
std::string talkToHardware(const std::string& command) {
    // 1. One-time terminal configuration to set baud rate and turn off echo
    static bool portConfigured = false;
    if (!portConfigured) {
        system("stty -F /dev/ttyACM0 115200 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");
        system("stty -F /dev/ttyACM1 115200 raw -echo -echoe -echok -echoctl -echoke 2>/dev/null");
        portConfigured = true;
    }

    // 2. Open the active interface mapping
    std::fstream serialPort("/dev/ttyACM0", std::ios::in | std::ios::out);
    if (!serialPort.is_open()) {
        serialPort.open("/dev/ttyACM1", std::ios::in | std::ios::out);
        if (!serialPort.is_open()) return "";
    }

    // 3. Dispatch the command and parse the line answer text
    serialPort << command << "\r\n";
    serialPort.flush();

    std::string response = "";
    std::getline(serialPort, response);
    return response;
}

// Inspects the state of physical Input 0
bool isPhysicalButtonPressed() {
    std::string state = talkToHardware("op dio-state");
    
    // Safety exit if communication drops out or string isn't populated
    if (state.length() < 8) return false;
    
    // Position 0 tracks Input 0.
    // Fixed: State is a string, so we check character index 0 against the character '0'
    return (state[0] == '0');
}

// Drives physical Output 0 high or low to change your physical lamp line
void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        talkToHardware("op set-output 0 1");
    } else {
        talkToHardware("op set-output 0 0");
    }
}

int main() {
    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon 400 Native IO Bridge");
    window.setFramerateLimit(30); 

    // Visual indicators inside the window
    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); 

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); 

    std::cout << "Application online. Click the visual window area to toggle the physical lamp." << std::endl;

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // WINDOW MOUSE INTERACTION: Updates physical lamp pin status
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "Lamp state toggled to: " << physicalLampState << std::endl;
                }
            }
        }

        // HARDWARE PIN SURVEY: Read physical button to toggle UI graphic colors
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); 
        } else {
            visualLamp.setFillColor(sf::Color::Red);   
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); 
    return 0;
}
