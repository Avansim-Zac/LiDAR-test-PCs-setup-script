#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>
#include <cstdlib>

// Helper to query the state of Input 0 using standard system interaction hooks
bool checkHardwareInput() {
    // We open a pipe to view the value of input pin zero
    // In0 is pulled High (1) by default; pulling it to Pin 2 (GND) drops the value to 0
    FILE* pipe = popen("cat /sys/class/gpio/gpio*/value 2>/dev/null", "r");
    if (!pipe) return false;

    char buffer[128];
    std::string result = "";
    while (fgets(buffer, sizeof(buffer), pipe) != NULL) {
        result += buffer;
    }
    pclose(pipe);

    // If the input reads back a low signal, the physical button is pressed
    return (result.find("0") != std::string::npos);
}

// Helper to write changes cleanly to Output 0 (Pin 4)
void toggleHardwareOutput(bool turnOn) {
    if (turnOn) {
        // Direct execution commands handled cleanly by the underlying PSE driver mapping
        std::system("echo 1 > /sys/class/gpio/gpio*/value 2>/dev/null");
    } else {
        std::system("echo 0 > /sys/class/gpio/gpio*/value 2>/dev/null");
    }
}

int main() {
    std::cout << "Loading Registered Driver Connection..." << std::endl;

    // Load the kernel driver instance explicitly to ensure communication paths are ready
    std::system("sudo modprobe pse 2>/dev/null");

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay IO Test");
    window.setFramerateLimit(20); // 20 FPS ensures low overhead on system pipes

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red);

    bool physicalLampState = false;
    toggleHardwareOutput(physicalLampState);

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // SCREEN INTERACTION EVENT: Toggles physical Output 0 (Pin 4)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    toggleHardwareOutput(physicalLampState);
                    std::cout << "Window Clicked! Pin 4 Set to: " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE EVENT SURVEY: Reads Pin 3 (In0) to update graphic colors dynamically
        if (checkHardwareInput()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical circuit closed
        } else {
            visualLamp.setFillColor(sf::Color::Red);   // Standby open circuit
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    toggleHardwareOutput(false); // Safeguard payload on close
    return 0;
}
