#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Raw firmware hex byte frames parsed natively by the OnLogic PSE ModBay controller
const uint8_t CMD_READ_DIO[]   = { 0x01, 0x00 }; 
const uint8_t CMD_WRITE_HIGH[] = { 0x02, 0x00, 0x01 }; // Set Out0 (Pin 4) On
const uint8_t CMD_WRITE_LOW[]  = { 0x02, 0x00, 0x00 }; // Set Out0 (Pin 4) Off

// Directly writes raw command byte sequences to your new driver node
bool toggleModBayOutput(const uint8_t* command, size_t size) {
    std::ofstream device("/dev/pse0", std::ios::binary);
    if (!device.is_open()) return false;
    
    device.write(reinterpret_cast<const char*>(command), size);
    device.flush();
    return true;
}

// Interrogates Pin 3 (In0) to see if it is bridged to Pin 2 (GND)
bool isPhysicalButtonPressed() {
    std::fstream device("/dev/pse0", std::ios::in | std::ios::out | std::ios::binary);
    if (!device.is_open()) return false;

    // Send a quick register poll request command
    device.write(reinterpret_cast<const char*>(CMD_READ_DIO), sizeof(CMD_READ_DIO));
    device.flush();

    // Catch the raw response byte returned from the motherboard driver
    char responseByte = 0;
    device.read(&responseByte, 1);

    // Bitwise check isolates index slot 0 (Input 0). 
    // It drops low (0) when your hardware button pulls the path to GND.
    return ((responseByte & 0x01) == 0);
}

void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        toggleModBayOutput(CMD_WRITE_HIGH, sizeof(CMD_WRITE_HIGH));
    } else {
        toggleModBayOutput(CMD_WRITE_LOW, sizeof(CMD_WRITE_LOW));
    }
}

int main() {
    std::cout << "Loading Integrated ModBay PSE Controller..." << std::endl;

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay Native IO Bridge");
    window.setFramerateLimit(30); // 30 FPS gives rapid, snappy hardware responses

    // Setup visual window elements
    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); // Default state: Off/Released

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Assure pins start clear on bootup

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // SCREEN CLICK EVENT: Left-clicking inside the window toggles Pin 4 (Out0)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "Dispatched Core Bus Signal. Lamp Pin 4 Target: " << (physicalLampState ? "HIGH" : "LOW") << std::endl;
                }
            }
        }

        // HARDWARE EVENT SURVEY: Survey Pin 3 (In0) to update window layouts dynamically
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical circuit closed
        } else {
            visualLamp.setFillColor(sf::Color::Red);   // Standby open circuit
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Resource cleanup safety sequence on application closure
    return 0;
}
