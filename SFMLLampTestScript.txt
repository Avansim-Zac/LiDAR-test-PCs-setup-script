#include <SFML/Graphics.hpp>
#include <iostream>
#include <fstream>
#include <string>

// Helper to write a command directly to the Karbon hardware port and read the response
std::string talkToHardware(const std::string& command) {
    // Open the raw serial device path used by OnLogic hardware
    std::fstream serialPort("/dev/ttyACM0", std::ios::in | std::ios::out);
    if (!serialPort.is_open()) {
        // Fallback to alternative port index if ACM0 isn't initialized
        serialPort.open("/dev/ttyACM1", std::ios::in | std::ios::out);
        if (!serialPort.is_open()) return "";
    }

    // Send the exact command string to the microcontroller
    serialPort << command << "\r\n";
    serialPort.flush();

    // Read the feedback line returned by the machine
    std::string response;
    std::getline(serialPort, response);
    return response;
}

// Queries the hardware button state
bool isPhysicalButtonPressed() {
    std::string state = talkToHardware("op dio-state");
    
    // If the port failed or returned a bad string, exit early
    if (state.length() < 8) return false;
    
    // Index 0 represents physical Input 0.
    // Pulled High (1) normally. Drops to Low (0) when the physical button links to GND.
    return (state[0] == '0');
}

// Controls the hardware lamp output pin
void setPhysicalLamp(bool turnOn) {
    if (turnOn) {
        talkToHardware("op set-output 0 1");
    } else {
        talkToHardware("op set-output 0 0");
    }
}

int main() {
    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon Native Port Test");
    window.setFramerateLimit(30); // 30 FPS provides a highly responsive UI

    // Setup visual interface indicators
    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red);

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState);

    std::cout << "Native app running. Click inside the window to toggle the physical lamp." << std::endl;

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // Mouse click event toggles the hardware lamp pin
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                }
            }
        }

        // Hardware Polling: Update window shape color depending on physical button
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green);
        } else {
            visualLamp.setFillColor(sf::Color::Red);
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Safeguard shutdown sequence
    return 0;
}
