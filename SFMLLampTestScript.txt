#include <SFML/Graphics.hpp>
#include <iostream>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

// Low-level Linux ioctl Magic Numbers defined by the OnLogic PSE kernel driver
#define PSE_IOCTL_MAGIC       'p'
#define PSE_IOCTL_GET_DI      _IOR(PSE_IOCTL_MAGIC, 1, uint8_t)
#define PSE_IOCTL_SET_DO      _IOW(PSE_IOCTL_MAGIC, 2, uint16_t)

// Queries Pin 3 (In0) using a native Linux ioctl call
bool isPhysicalButtonPressed() {
    // Open the driver file in read-only mode
    int fd = open("/dev/pse0", O_RDONLY);
    if (fd < 0) return false;

    uint8_t inputStates = 0;
    // Execute the ioctl command to pull the current hardware registers
    if (ioctl(fd, PSE_IOCTL_GET_DI, &inputStates) < 0) {
        close(fd);
        return false;
    }
    close(fd);

    // Bit 0 tracks Input 0 (Pin 3). It drops to '0' when shorted to Pin 2 (GND).
    return ((inputStates & 0x01) == 0);
}

// Controls Pin 4 (Out0) using a native Linux ioctl call
void setPhysicalLamp(bool turnOn) {
    int fd = open("/dev/pse0", O_WRONLY);
    if (fd < 0) return;

    // The driver expects a 16-bit integer where:
    // High byte = Pin Index (0 for Output 0)
    // Low byte = State (1 for High, 0 for Low)
    uint16_t payload = (0 << 8) | (turnOn ? 1 : 0);

    // Execute the ioctl command to switch the physical transistor line
    ioctl(fd, PSE_IOCTL_SET_DO, &payload);
    close(fd);
}

int main() {
    std::cout << "Loading Native Linux ioctl Driver Interface..." << std::endl;

    sf::RenderWindow window(sf::VideoMode(400, 400), "Karbon ModBay IO Test");
    window.setFramerateLimit(30); 

    sf::CircleShape visualLamp(70.f);
    visualLamp.setOrigin(70.f, 70.f);
    visualLamp.setPosition(200.f, 200.f);
    visualLamp.setFillColor(sf::Color::Red); 

    bool physicalLampState = false;
    setPhysicalLamp(physicalLampState); // Turn the physical lamp off on start

    while (window.isOpen()) {
        sf::Event event;
        while (window.pollEvent(event)) {
            if (event.type == sf::Event::Closed) {
                window.close();
            }
            
            // SCREEN CLICK EVENT: Left-clicking inside the window toggles Pin 4 (Out0)
            if (event.type == sf::Event::MouseButtonPressed) {
                if (event.mouseButton.button == sf::Mouse::Left) {
                    physicalLampState = !physicalLampState;
                    setPhysicalLamp(physicalLampState);
                    std::cout << "UI Mouse Click. Output State -> " << (physicalLampState ? "ON" : "OFF") << std::endl;
                }
            }
        }

        // HARDWARE POLLING LOOP: Reads Pin 3 (In0) to update graphic colors dynamically
        if (isPhysicalButtonPressed()) {
            visualLamp.setFillColor(sf::Color::Green); // Physical button pressed
        } else {
            visualLamp.setFillColor(sf::Color::Red);   // Physical button released
        }

        window.clear(sf::Color(30, 30, 30));
        window.draw(visualLamp);
        window.display();
    }

    setPhysicalLamp(false); // Safeguard shutdown sequence
    return 0;
}
